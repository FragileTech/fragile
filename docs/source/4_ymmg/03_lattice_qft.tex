\begin{abstract}
This document constructs a finite and explicit $SU(3)$ lattice gauge structure from the
interaction graph generated by the viscous force channel of the interacticle run.
No axiom verification is performed. Every definition needed for a later verification
paper is introduced and proved here.
\end{abstract}

\section{Imported facts and notation}
\begin{definition}[External assumptions from \texttt{01\_convergence.tex}]
\label{def:imports}
Let $d=3$. Fix kernel parameters
\[
\ell_{\mathrm{visc}}>0,\qquad \kappa_0>0,\qquad \nu_{\mathrm{visc}}\ge0,
\]
and
\[
K_\rho(x,y):=\kappa_0+\exp\!\left(-\frac{\|x-y\|^2}{2\ell_{\mathrm{visc}}^2}\right),\qquad \rho:=\ell_{\mathrm{visc}}.
\]
From the kernel assumption in \texttt{01\_convergence.tex}, the kernel satisfies
\[
\kappa_*:=\inf_{x,y}K_\rho(x,y)=\kappa_0,\qquad
\kappa^*:=\sup_{x,y}K_\rho(x,y)=1+\kappa_0<\infty.
\]
No other assumptions from the kinetic convergence proof are imported.
\medskip

A recorded trajectory is available at discrete times
\[
\Tset=\{0,1,\dots,T\},\quad T\in\N,
\]
with finite alive sets $\mathcal A(t)\subseteq\{1,\ldots,N\}$ and finite node data
\[
x_i(t),\,v_i(t)\in\R^3,\qquad \Phi_i(t)\in\R,\qquad i\in\mathcal A(t),\ t\in\Tset.
\]
At each timestep $t$, $\mathcal P_t\subseteq \mathcal A(t)\times\mathcal A(t)\setminus\{(i,i):i\in\mathcal A(t)\}$
is finite and records the ordered interaction companions used by the algorithm at $t$.
\end{definition}

\begin{lemma}[Kernel orthogonal symmetry]
\label{lem:kernel-orth}
For every orthogonal matrix $O\in O(3)$ and all $x,y\in\R^3$,
\[
K_\rho(Ox,Oy)=K_\rho(x,y).
\]
\end{lemma}
\begin{proof}
By definition,
\[
\|Ox-Oy\|^2=\|x-y\|^2,
\]
so the Gaussian factor is invariant under simultaneous orthogonal transformation.
\end{proof}

\begin{lemma}[Orthogonal covariance]
\label{lem:orthogonal}
Let $O\in O(3)$ and define transformed data $x_i'(t)=Ox_i(t)$, $v_i'(t)=Ov_i(t)$.
Then for each node $x=n_{i,t}$,
\[
\mathbf F^{\mathrm{visc}}(x;\text{transformed data})=O\,\mathbf F^{\mathrm{visc}}(x;\text{original data}).
\]
\end{lemma}
\begin{proof}
Lemma~\ref{lem:kernel-orth} gives $K_\rho(Ox_i,Ox_j)=K_\rho(x_i,x_j)$, hence
\[
\deg_{\mathrm{IG}}'(i,t)=\deg_{\mathrm{IG}}(i,t),
\qquad
\omega_{ij}^{\mathrm{IG}\prime}(t)=\omega_{ij}^{\mathrm{IG}}(t).
\]
For each $e=(n_{i,t}\to n_{j,t})$,
\[
\mathbf F^{\mathrm{visc}}{}'(e)
=\nu_{\mathrm{visc}}\,\omega^{\mathrm{IG}}_{ij}(t)(Ov_j-Ov_i)
=O\,\mathbf F^{\mathrm{visc}}(e).
\]
Summing outgoing IG edges at a fixed source node preserves left multiplication by $O$.
\end{proof}

\section{Recorded run and interaction complex}
\label{sec:run}
\begin{definition}[Recorded interacticle run]
\label{def:run}
The recorded run is the tuple
\[
\mathscr R=\Bigl(T,\{\mathcal A(t)\}_{t\in\Tset},\{\mathcal P_t\}_{t\in\Tset},\{x_i(t),v_i(t),\Phi_i(t)\},\{K_\rho\}\Bigr).
\]
\end{definition}

\begin{definition}[Node set and projections]
\label{def:nodes}
For each alive pair $(i,t)$ define a spacetime node $n_{i,t}$ and set
\[
\mcN:=\{n_{i,t}\mid i\in\mathcal A(t),\ t\in\Tset\}.
\]
Define spatial and velocity projections
\[
\mathbf x(n_{i,t})=x_i(t),\qquad \mathbf v(n_{i,t})=v_i(t).
\]
Both maps are well-defined by definition of $\mcN$.
\end{definition}

\begin{definition}[Oriented edge sets]
\label{def:edges}
Define
\[
E_{\mathrm{CST}}:=\{(n_{i,t}\to n_{i,t+1}):\,i\in\mathcal A(t),\ t=0,\dots,T-1\},
\]
\[
E_{\mathrm{IG}}:=\{(n_{i,t}\to n_{j,t}):\,(i,j)\in\mathcal P_t,\ t\in\Tset\},
\]
\[
E_{\mathrm{IA}}:=\{(n_{i,t+1}\to n_{j,t}):\,(i,j)\in\mathcal P_t,\ t=0,\dots,T-1\}.
\]
Set
\[
\mcE:=E_{\mathrm{CST}}\cup E_{\mathrm{IG}}\cup E_{\mathrm{IA}}.
\]
For $e=(u\to v)\in\mcE$, write $s(e):=u$ and $t(e):=v$, and define $e^{-1}:=(t(e)\to s(e))$.
\end{definition}

\begin{definition}[Interaction triangles and plaquettes]
\label{def:triangles}
For each $(i,j)\in\mathcal P_t$ with $t<T$, define the oriented interaction triangle
\[
\triangle_{ij,t}:=(n_{i,t},n_{i,t+1},n_{j,t}).
\]
Its oriented boundary is
\[
\partial\triangle_{ij,t}=e_{\mathrm{CST}}(n_{i,t}\to n_{i,t+1})
+e_{\mathrm{IA}}(n_{i,t+1}\to n_{j,t})-e_{\mathrm{IG}}(n_{i,t}\to n_{j,t}).
\]
If both ordered pairs $(i,j),(j,i)\in\mathcal P_t$ with $t<T$, define
\[
P_{ij,t}:=\triangle_{ij,t}\cup \triangle_{ji,t}.
\]
Its boundary is the 4-cycle
\[
\partial P_{ij,t}=e_{\mathrm{CST}}(n_{i,t}\to n_{i,t+1})
+e_{\mathrm{IA}}(n_{i,t+1}\to n_{j,t})
+e_{\mathrm{CST}}(n_{j,t}\to n_{j,t+1})
+e_{\mathrm{IA}}(n_{j,t+1}\to n_{i,t}).
\]
Set
\[
\mcT:=\{\triangle_{ij,t}\mid (i,j)\in\mathcal P_t,\ t=0,\dots,T-1\},\qquad
\mathcal F:=(\mcN,\mcE,\mcT).
\]
\[
\bar E_{\mathrm{CST}}:=\{\{u,v\}:(u\to v)\in E_{\mathrm{CST}}\},\quad
\bar E_{\mathrm{IG}}:=\{\{u,v\}:(u\to v)\in E_{\mathrm{IG}}\},\quad
\bar E_{\mathrm{IA}}:=\{\{u,v\}:(u\to v)\in E_{\mathrm{IA}}\},
\]
\[
\bar E:=\bar E_{\mathrm{CST}}\cup\bar E_{\mathrm{IG}}\cup\bar E_{\mathrm{IA}},\qquad
\mathcal F_{\mathrm{und}}:=(\mcN,\bar E,\mcT).
\]
\end{definition}

\begin{proposition}[Directed 2-complex structure]
\label{prop:complex}
All faces and boundary edges of $\mathcal F_{\mathrm{und}}$ are finite.
For each oriented edge $e=(u\to v)\in\mcE$ define \(\partial_1e=v-u\), and for each triangle \(\triangle_{ij,t}\) define
\[
\partial_2\triangle_{ij,t}
=e_{\mathrm{CST}}(n_{i,t}\to n_{i,t+1})
+e_{\mathrm{IA}}(n_{i,t+1}\to n_{j,t})
-e_{\mathrm{IG}}(n_{i,t}\to n_{j,t}).
\]
For each $t<T$ with $(i,j)\in\mathcal P_t$,
\[
\partial P_{ij,t}=\partial\triangle_{ij,t}+\partial\triangle_{ji,t},
\]
where the IG contributions cancel by opposite orientations.
\end{proposition}
\begin{proof}
Each face and boundary edge is explicitly built from finite sets in Definitions~\ref{def:run}--\ref{def:triangles}.
\(\partial_2\) is the oriented boundary formula in Definition~\ref{def:triangles}.
In the sum defining \(\partial P_{ij,t}\), each of the two interaction triangles contributes an IG edge with opposite sign; they cancel.
\end{proof}

\begin{proposition}[Finiteness and cardinality]
\label{prop:finiteness}
All sets in Definitions~\ref{def:run}--\ref{def:triangles} are finite.
\[
E_{\mathrm{IG}}^{<T}:=\{(n_{i,t}\to n_{j,t}):\,(i,j)\in\mathcal P_t,\ t\in\{0,\dots,T-1\}\}
\]
For $m_t:=|\mathcal P_t|$,
\[
|E_{\mathrm{CST}}|=\sum_{t=0}^{T-1}|\mathcal A(t)|,\qquad
|E_{\mathrm{IG}}^{<T}|=|\mcT|=|E_{\mathrm{IA}}|=\sum_{t=0}^{T-1}m_t,\qquad
|E_{\mathrm{IG}}|=\sum_{t=0}^{T}m_t.
\]
For each $t<T$ with $(i,j)\in\mathcal P_t$,
\[
\partial P_{ij,t}=\partial\triangle_{ij,t}+\partial\triangle_{ji,t},
\]
with the shared edge contribution canceling because
\[
e_{\mathrm{IG}}(n_{i,t}\to n_{j,t})=-\,e_{\mathrm{IG}}(n_{j,t}\to n_{i,t})
\]
in the oriented sum.
\end{proposition}
\begin{proof}
Finiteness follows from finiteness of $\Tset$, all $\mathcal A(t)$ and all $\mathcal P_t$, together with the set definitions.
Cardinality identities are by disjoint union over timesteps and one-to-one correspondence between
$\mathcal P_t$ and contributions in $E_{\mathrm{IG}}$, $E_{\mathrm{IA}}$, and $\mcT$.
The boundary identity is direct from the explicit formulas in Definition~\ref{def:triangles}.
\end{proof}

\section{Viscous force from interaction edges}
\begin{definition}[IG degree and pairwise force coefficients]
\label{def:ig-weights}
For each $(i,t)$ define the raw degree and normalized IG coefficients
\[
\deg_{\mathrm{IG}}(i,t):=\sum_{(i,j)\in\mathcal P_t}K_\rho(x_i(t),x_j(t)),
\]
\[
\omega^{\mathrm{IG}}_{ij}(t):=
\begin{cases}
\dfrac{K_\rho(x_i(t),x_j(t))}{\deg_{\mathrm{IG}}(i,t)},&(i,j)\in\mathcal P_t,\ \deg_{\mathrm{IG}}(i,t)>0,\\[0.3em]
0,&\text{otherwise.}
\end{cases}
\]
\end{definition}

\begin{definition}[Viscous pair force and node force]
\label{def:forces}
For each edge $e=(n_{i,t}\to n_{j,t})\in E_{\mathrm{IG}}$ define
\[
\mathbf F^{\mathrm{visc}}(e):=\nu_{\mathrm{visc}}\omega^{\mathrm{IG}}_{ij}(t)\,(v_j(t)-v_i(t))\in\R^3.
\]
For $x=n_{i,t}\in\mcN$ define
\[
\mathbf F^{\mathrm{visc}}(x):=\sum_{\substack{e\in E_{\mathrm{IG}}\\ s(e)=x}}\mathbf F^{\mathrm{visc}}(e).
\]
If the sum is over an empty set, it is $0$.
\end{definition}

\begin{lemma}[Weight properties]
\label{lem:weights}
For every $i\in\mathcal A(t)$ and $t\in\Tset$,
\[
\omega^{\mathrm{IG}}_{ij}(t)\ge0,\qquad
\sum_{j:(i,j)\in\mathcal P_t}\omega^{\mathrm{IG}}_{ij}(t)=\mathbf 1_{\{\deg_{\mathrm{IG}}(i,t)>0\}}.
\]
\end{lemma}
\begin{proof}
Positivity is immediate from $K_\rho\ge\kappa_*>0$.
If $\deg_{\mathrm{IG}}(i,t)>0$, summing the definitions over $j$ gives
\[
\sum_{j:(i,j)\in\mathcal P_t}\omega^{\mathrm{IG}}_{ij}(t)
=\frac{1}{\deg_{\mathrm{IG}}(i,t)}\sum_{j:(i,j)\in\mathcal P_t}K_\rho(x_i(t),x_j(t))=1.
\]
If $\deg_{\mathrm{IG}}(i,t)=0$, every summand in Definition~\ref{def:ig-weights} is $0$
by convention and the sum is $0$.
\end{proof}

\begin{lemma}[Node reconstruction of viscous force]
\label{lem:reconstruction}
For every $x=n_{i,t}\in\mcN$,
\[
\mathbf F^{\mathrm{visc}}(x)=\sum_{\substack{e\in\mcE\\ s(e)=x,\ e\in E_{\mathrm{IG}}}}\mathbf F^{\mathrm{visc}}(e).
\]
\end{lemma}
\begin{proof}
By Definition of $E_{\mathrm{IG}}$, the summands are exactly the ordered companions $(i,j)\in\mathcal P_t$
with source $n_{i,t}=x$. This is the same index set used in Definition~\ref{def:forces}.
\end{proof}

\begin{proposition}[Orthogonal basis covariance]
\label{thm:viscous-su3}
Let $O\in O(3)$ and consider transformed data
\[
x_i'(t)=Ox_i(t),\qquad v_i'(t)=Ov_i(t).
\]
For every node $x\in\mcN$,
\[
\mathbf F^{\mathrm{visc}}(x;\text{transformed data})=O\,\mathbf F^{\mathrm{visc}}(x;\text{data}).
\]
Real matrices satisfy $O^\dagger=O^\top$, hence $O\in U(3)$, so orthogonal force-frame
changes embed in unitary changes of complexified color index.
\end{proposition}
\begin{proof}
From Lemma~\ref{lem:orthogonal}, the kernel-based force coefficients are unchanged by
orthogonal rotation of $(x,v)$. Real entries give $O^\dagger=O^\top$, so $O^\dagger O=I$ and
$O$ acts unitarily on $\C^3$.
\end{proof}

\section{Color triplets and $SU(3)$ edge transport}
\label{sec:su3}

\begin{definition}[Normalized color vectors]
\label{def:colors}
Fix constants $g_3>0,\ \ell_0>0,\ \hbar_{\mathrm{eff}}>0,\ \mu_p>0,\ \epsilon_c>0$.
For $x=n_{i,t}\in\mcN$, define
\[
\tilde c_x^{(\alpha)}:=(\mathbf F^{\mathrm{visc}}(x))_\alpha
\exp\!\left(i\frac{\mu_p\,v_i^{(\alpha)}(t)\ell_0}{\hbar_{\mathrm{eff}}}\right),\quad \alpha=1,2,3,
\]
\[
\mathbf c_x:=\frac{\tilde{\mathbf c}_x}{\sqrt{\tilde{\mathbf c}_x^\dagger\tilde{\mathbf c}_x+\epsilon_c^2}}\in\C^3.
\]
Then $\mathbf c_x^\dagger\mathbf c_x\in[0,1)$ and $\mathbf c_x$ is defined for all $x\in\mcN$.
\end{definition}
\begin{lemma}[Normalization of color vectors]
\label{lem:color-norm}
For every node $x\in\mcN$,
\[
\mathbf c_x^\dagger\mathbf c_x
=\frac{\tilde{\mathbf c}_x^\dagger\tilde{\mathbf c}_x}{\tilde{\mathbf c}_x^\dagger\tilde{\mathbf c}_x+\epsilon_c^2}\in[0,1),
\]
so $\mathbf c_x$ is well defined and bounded.
\end{lemma}
\begin{proof}
By construction $\epsilon_c>0$, so $\tilde{\mathbf c}_x^\dagger\tilde{\mathbf c}_x+\epsilon_c^2>0$.
Therefore the formula is well defined and gives the stated ratio.
\end{proof}
\begin{proposition}[Local $SU(3)$ gauge reparametrization]
\label{prop:su3-redundancy}
Let $\mathbf c_x$ be defined in Definition~\ref{def:colors} and let
\[
\Omega:\mcN\to SU(3)
\]
be arbitrary.
Define transformed data by
\[
\mathbf c_x^\Omega:=\Omega(x)\mathbf c_x,\qquad
U_3^\Omega(e):=\Omega(t(e))\,U_3(e)\,\Omega(s(e))^{-1},\quad e=(s(e)\to t(e))\in\mcE.
\]
For every $x\in\mcN$ and every $e\in\mcE$,
\[
(\mathbf c_x^\Omega)^\dagger\mathbf c_x^\Omega=\mathbf c_x^\dagger\mathbf c_x,\qquad
U_3^\Omega(e)\in SU(3).
\]
This is the local $SU(3)$ color redundancy of the viscous interaction complex.
\end{proposition}
\begin{proof}
For the norm, $\Omega(x)^\dagger\Omega(x)=I$.
Proposition~\ref{prop:su3-edge} gives $U_3(e)\in SU(3)$, and $SU(3)$ is closed under
conjugation by elements of $SU(3)$.
\end{proof}

\begin{definition}[Gell--Mann generators]
\label{def:gellmann}
Let $\lambda_a\in\C^{3\times3}$, $a=1,\dots,8$, be
\[
\lambda_1=\begin{pmatrix}0&1&0\\1&0&0\\0&0&0\end{pmatrix},\;
\lambda_2=\begin{pmatrix}0&-i&0\\ i&0&0\\0&0&0\end{pmatrix},\;
\lambda_3=\begin{pmatrix}1&0&0\\0&-1&0\\0&0&0\end{pmatrix},
\]
\[
\lambda_4=\begin{pmatrix}0&0&1\\0&0&0\\1&0&0\end{pmatrix},\;
\lambda_5=\begin{pmatrix}0&0&-i\\0&0&0\\ i&0&0\end{pmatrix},\;
\lambda_6=\begin{pmatrix}0&0&0\\0&0&1\\0&1&0\end{pmatrix},
\]
\[
\lambda_7=\begin{pmatrix}0&0&0\\0&0&-i\\0&i&0\end{pmatrix},\;
\lambda_8=\frac1{\sqrt3}\begin{pmatrix}1&0&0\\0&1&0\\0&0&-2\end{pmatrix}.
\]
Each $\lambda_a$ is Hermitian and $\operatorname{tr}\lambda_a=0$.
\end{definition}

\begin{lemma}[Gell--Mann identities]
\label{lem:gellmann}
These matrices satisfy
\[
\tr(\lambda_a\lambda_b)=2\delta_{ab},\qquad a,b\in\{1,\dots,8\}.
\]
\end{lemma}
\begin{proof}
Direct matrix multiplication gives diagonal quadratic terms $2$ for $a=b$
and zero trace for $a\ne b$. The full check is finite and componentwise.
\end{proof}

\begin{lemma}[Hermitian traceless span]
\label{lem:hermitian-span}
If $\alpha_1,\dots,\alpha_8\in\R$, then
\[
H:=\sum_{a=1}^8\alpha_a\lambda_a
\]
is Hermitian and trace free.
\end{lemma}
\begin{proof}
Each $\lambda_a$ is Hermitian and trace free by Definition~\ref{def:gellmann}.
Real linear combinations preserve Hermiticity and zero trace.
\end{proof}

\begin{definition}[Generator and transport on edges]
\label{def:transport}
For each oriented edge $e=(x\to y)\in\mcE$ define
\[
\Theta_e:=\frac{g_3}{2}\sum_{a=1}^8 \Re\!\bigl(\mathbf c_x^\dagger\lambda_a\mathbf c_y\bigr)\lambda_a\in\{X\in\C^{3\times3}:X^\dagger=X,\ \tr X=0\},
\]
\[
U_3(e):=\exp(i\Theta_e)\in SU(3),
\]
and $U_3(e^{-1}):=U_3(e)^{-1}$.
\end{definition}

\begin{proposition}[Edge transport is $SU(3)$]
\label{prop:su3-edge}
For every $e\in\mcE$, $U_3(e)\in SU(3)$ and $U_3(e^{-1})=U_3(e)^{-1}$.
\end{proposition}
\begin{proof}
Each coefficient $\Re(\mathbf c_x^\dagger\lambda_a\mathbf c_y)$ is real.
By Lemma~\ref{lem:hermitian-span}, $\Theta_e$ is Hermitian and trace free, so
\(i\Theta_e\) is skew-Hermitian and trace free.
Hence \(\exp(i\Theta_e)\) is unitary.
Because $\tr\Theta_e=0$, $\det U_3(e)=\exp(i\tr\Theta_e)=1$.
The relation for inverse edges is forced by the convention $U_3(e^{-1}):=U_3(e)^{-1}$.
\end{proof}
\begin{remark}
When $x=y$ and $e$ is a self-oriented edge, $\Theta_e$ is still well-defined and $U_3(e)$ may be nontrivial.
\end{remark}

\section{Holonomy and gauge invariance}
\begin{definition}[Path holonomy]
\label{def:path}
For an oriented path
\[
\gamma=(v_0\to v_1,\dots,v_{m-1}\to v_m),\qquad v_k\in\mcN,
\]
define
\[
U_3(\gamma):=\prod_{k=1}^m U_3(v_{k-1}\to v_k),
\]
ordered from $k=1$ to $k=m$.
If $v_m=v_0$, $\gamma$ is a loop and $W_3(\gamma):=\tr U_3(\gamma)$.
\end{definition}

\begin{definition}[Triangle and plaquette holonomy]
\label{def:holonomy}
For a triangle or plaquette with oriented boundary edges $\partial\triangle$ or $\partial P$,
\[
U_3(\triangle):=\prod_{e\in\partial\triangle}U_3(e),\qquad
U_3(P):=\prod_{e\in\partial P}U_3(e),
\]
with the order following the oriented boundary sequence.
\end{definition}

\begin{proposition}[Plaquette factorization]
\label{prop:plaquette-factorization}
For each plaquette $P_{ij,t}$,
\[
U_3(P_{ij,t})=U_3(\triangle_{ij,t})\,U_3(\triangle_{ji,t}).
\]
\end{proposition}
\begin{proof}
By definition $\partial P_{ij,t}=\partial\triangle_{ij,t}+\partial\triangle_{ji,t}$, and the two IG edge factors are
with opposite orientation. By $U_3(e^{-1})=U_3(e)^{-1}$ the shared factor reduces to identity.
\end{proof}
\begin{remark}
No additional assumptions are used: the result is direct from edge orientation conventions in Definitions~\ref{def:edges} and~\ref{def:triangles}.
\end{remark}

\begin{definition}[Gauge map]
\label{def:gauge}
A gauge map is $\Omega:\mcN\to SU(3)$.
For an edge $e=(x\to y)$ define
\[
U_3^\Omega(e):=\Omega(y)\,U_3(e)\,\Omega(x)^{-1}.
\]
Define transformed color vectors by $\mathbf c_x^\Omega:=\Omega(x)\mathbf c_x$.
\end{definition}

\begin{proposition}[Gauge invariance of Wilson traces]
\label{prop:gauge-invariance}
For any closed loop $\gamma=(v_0,\dots,v_m=v_0)$,
\[
\tr\left(U_3^\Omega(\gamma)\right)=\tr\left(U_3(\gamma)\right).
\]
\end{proposition}
\begin{proof}
Write $U_3^\Omega(\gamma)=\prod_{k=1}^m\Omega(v_k)U_3(v_{k-1}\to v_k)\Omega(v_{k-1})^{-1}$.
Consecutive factors telescope to
\[
U_3^\Omega(\gamma)=\Omega(v_0)\,U_3(\gamma)\,\Omega(v_0)^{-1}.
\]
Trace cyclicity removes the conjugation.
\end{proof}

\begin{definition}[Wilson action]
\label{def:action}
Let
\[
\mathcal P_\square:=\{P_{ij,t}\mid (i,j),(j,i)\in\mathcal P_t,\ t=0,\dots,T-1\},
\qquad
\beta:=\frac{6}{g_3^2}.
\]
Define
\[
S_{\mathrm W}:=\beta\sum_{P\in\mathcal P_\square}\left(1-\frac13\Re\tr U_3(P)\right).
\]
\end{definition}

\begin{proposition}[Gauge invariance of Wilson action]
\label{prop:wilson-gauge}
For any gauge map $\Omega$, $S_{\mathrm W}^\Omega=S_{\mathrm W}$.
\end{proposition}
\begin{proof}
Each term in the sum is invariant by Proposition~\ref{prop:gauge-invariance}.
Therefore $S_{\mathrm W}$ is invariant.
\end{proof}

\section{Matter sector on the same complex}\label{sec:matter}
\begin{definition}[Matter fields and actions]
\label{def:matter}
For each node $x\in\mcN$ choose
\[
\psi_x\in\C^3,\qquad \bar\psi_x\in(\C^3)^\dagger,\qquad \phi_x\in\R.
\]
Fix $m_F>0,\ m_\phi>0$ and a local potential $V_\phi:\R\to\R$.
For $e=(x\to y)\in\mcE$ set $\ell_{xy}:=\|\mathbf x(y)-\mathbf x(x)\|+\eta$ with $\eta>0$.
\[
S_{\mathrm F}:=\sum_{e\in E_{\mathrm{CST}}\cup E_{\mathrm{IA}}}\bar\psi_{t(e)}\left(\psi_{t(e)}-U_3(e)\psi_{s(e)}\right)
+m_F\sum_{x\in\mcN}\bar\psi_x\psi_x,
\]
\[
S_\phi:=\frac12\sum_{e\in E_{\mathrm{CST}}\cup E_{\mathrm{IG}}}\frac{(\phi_{t(e)}-\phi_{s(e)})^2}{\ell_{xy}}
+\sum_{x\in\mcN}\Bigl(\frac{m_\phi^2}{2}\phi_x^2+V_\phi(\phi_x)\Bigr).
\]
Gauge action on matter is
\[
\psi_x\mapsto\Omega(x)\psi_x,\qquad
\bar\psi_x\mapsto\bar\psi_x\Omega(x)^{-1},\qquad
\phi_x\mapsto\phi_x.
\]
\end{definition}

\begin{proposition}[Matter gauge invariance]
\label{prop:matter-gauge}
Under this gauge action,
\[
S_{\mathrm F}^\Omega=S_{\mathrm F},\qquad S_\phi^\Omega=S_\phi.
\]
\end{proposition}
\begin{proof}
For each edge term,
\[
\bar\psi_{t(e)}\left(\psi_{t(e)}-U_3(e)\psi_{s(e)}\right)
\mapsto
\bar\psi_{t(e)}\Omega(t(e))^{-1}
\left(\Omega(t(e))\psi_{t(e)}-\Omega(t(e))U_3(e)\Omega(s(e))^{-1}\Omega(s(e))\psi_{s(e)}\right),
\]
which is equal to the original term.
The mass term $\bar\psi_x\psi_x$ is invariant because $\Omega^{-1}\Omega=I$ pointwise.
$S_\phi$ is built from scalars and is unchanged.
\end{proof}
\begin{remark}
Only $E_{\mathrm{CST}}\cup E_{\mathrm{IA}}$ and $E_{\mathrm{CST}}\cup E_{\mathrm{IG}}$ appear in the matter actions.

\section{Finite lattice-QFT data package}
\begin{theorem}[Finite $SU(3)$ data object]
\label{thm:package}
Given a run as in Definition~\ref{def:run} and the constructions
in Sections~\ref{sec:run}--\ref{sec:matter}, the finite tuple
\[
\mathcal Q=
\Bigl(
\begin{aligned}
T,\{\mathcal A(t)\},\{\mathcal P_t\},\mcN,\mcE,\mcT,\\
\{x_i(t),v_i(t),\Phi_i(t)\}_{i,t},K_\rho,\nu_{\mathrm{visc}},
	\hbar_{\mathrm{eff}},\mu_p,\ell_0,\epsilon_c,g_3,\beta,\eta,\\
\{U_3(e)\}_{e\in\mcE},S_{\mathrm W},S_{\mathrm F},S_\phi
\end{aligned}
\Bigr)
\]
is explicit and finite.
It carries:
\begin{enumerate}[leftmargin=1.5em]
\item the directed interaction complex $(\mcN,\mcE,\mcT)$;
\item $SU(3)$ transport maps on edges;
\item Wilson traces and Wilson action;
\item gauge action on fermionic and scalar sectors.
\end{enumerate}
All objects needed for a subsequent axiom-verification paper are available.
\end{theorem}
\begin{proof}
Finiteness of all sets and sums follows from Proposition~\ref{prop:finiteness}
and the finiteness of the run data.
All remaining constructions are finite products and sums in finite dimensions.
\end{proof}
